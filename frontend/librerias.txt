# FASE 3.2: Instalación de dependencias principales
# ================================================

# 🚀 ROUTING
npm install react-router-dom

# 🎨 UI Y ESTILOS
# Instalar Tailwind CSS
npm install -D tailwindcss postcss autoprefixer
npx tailwindcss init -p

# Componentes UI y iconos
npm install @headlessui/react @heroicons/react lucide-react

# 🗂️ ESTADO GLOBAL
npm install zustand

# 📝 FORMULARIOS
npm install react-hook-form @hookform/resolvers yup

# 🌐 HTTP CLIENT
npm install axios

# 📊 GRÁFICOS Y CHARTS
npm install recharts

# 📋 TABLAS DE DATOS
npm install @tanstack/react-table

# 📅 FECHAS
npm install date-fns

# 🔔 NOTIFICACIONES
npm install react-hot-toast

# ⏳ LOADING Y SPINNERS
npm install react-spinners

# 🧰 UTILIDADES
npm install clsx lodash

# 🧪 TESTING
npm install -D @testing-library/react @testing-library/jest-dom @testing-library/user-event

# 📄 PDF GENERATION (opcional para reportes)
npm install jspdf html2canvas

# 📊 EXCEL EXPORT (opcional para reportes)
npm install exceljs
# y sheel excel
npm install https://cdn.sheetjs.com/xlsx-0.20.2/xlsx-0.20.2.tgz

# 🎯 DRAG AND DROP (opcional para interfaces avanzadas)
npm install @hello-pangea/dnd
